DROP TABLE Book;

CREATE TABLE Book
(
ID NUMBER(4) NOT NULL,
TITLE VARCHAR2(100),
AUTHOR VARCHAR2(50),
BOOK_FORMAT VARCHAR2(15),
LANGUAGE VARCHAR2(15),
AUDIENCE VARCHAR2(30),
PAGE_NUMBER VARCHAR2(10),
DATE_PUBLISHED VARCHAR2(30),
PUBLISHER VARCHAR2(50),
PUBLISHER_COUNTRY VARCHAR2(25),
DIMENSIONS VARCHAR2(15),
WEIGHT VARCHAR2(15),
PRICE FLOAT(6),
Primary key (ID)
);

commit;
